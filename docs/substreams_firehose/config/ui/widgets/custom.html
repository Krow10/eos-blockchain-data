 23def colorize_256(text: str, default_color: int = 0) -> list[tuple[int, int]]:
 24	"""
 25	Parses a string containing ANSI code escape sequences for `curses` display.
 26
 27	The ANSI code supported are the ones defined for the 256 color palette (see [here](
 28		https://gist.github.com/fnky/458719343aabd01cfb17a3a4f7296797#256-colors
 29	) for reference). Attributes such as bold, italic, etc. are also supported except for the strikethrough attribute.
 30
 31	This is used internally by the `CodeHighlightedPager` for coloring text.
 32
 33	*Note: background colors currently don't work.*
 34
 35	Args:
 36		text: The text to colorize.
 37		default_color: A default color for the *reset* escape sequences.
 38
 39	Returns:
 40		A list of tuples containing the `curses` attribute value and the number of character for which it applies.
 41	"""
 42	# Create the color map only once
 43	if not hasattr(colorize_256, 'color_map'):
 44		curses.use_default_colors()
 45
 46		colorize_256.color_map = {
 47			'39': default_color,
 48			'49': default_color,
 49			'39;49': default_color
 50		}
 51
 52		# TODO: Figure out how to make backgrounds work
 53		for i in range(curses.COLORS):
 54			for j in range(curses.COLORS):
 55				curses.init_pair(i + j * curses.COLORS + 1, i, j - 1)
 56				colorize_256.color_map[f'38;5;{i};48;5;{j}'] = curses.color_pair(i + j * curses.COLORS + 1)
 57			colorize_256.color_map[f'38;5;{i}'] = curses.color_pair(i + 1)
 58			colorize_256.color_map[f'48;5;{i}'] = curses.color_pair(i * curses.COLORS + 1)
 59
 60	# No escape code detected at all so print all text with the default color
 61	if not '\x1b[' in text:
 62		return [(default_color, len(text))]
 63
 64	colors = []
 65	# See https://gist.github.com/fnky/458719343aabd01cfb17a3a4f7296797#colors--graphics-mode for reference
 66	attr = {
 67		"00": curses.A_NORMAL,
 68		"01": curses.A_BOLD,
 69		"02": curses.A_DIM,
 70		"03": curses.A_ITALIC,
 71		"04": curses.A_UNDERLINE,
 72		"05": curses.A_BLINK,
 73		"07": curses.A_REVERSE,
 74		"08": curses.A_INVIS,
 75		# `ESC[9m` strikethrough is not available in Python curses (see https://docs.python.org/3/library/curses.html#curses.ncurses_version)
 76		"09": curses.A_NORMAL
 77	}
 78
 79	for substring in [sub for sub in text.split('\x1b[') if sub]:
 80		try:
 81			ansi_code, substring_text = substring.split('m', 1)
 82		except ValueError:
 83			# No ANSI code detected
 84			colors.append((default_color, len(substring)))
 85			continue
 86
 87		text_attribute = 0
 88
 89		# Validate ANSI code format or else its just plain text that happens to have an 'm' in it.
 90		if not all(code.isdigit() for code in ansi_code.split(';')):
 91			colors.append((default_color, len(substring_text)))
 92			continue
 93
 94		codes = ansi_code.split(';')
 95		ansi_code = ''
 96
 97		# Parses special attributes (bold, etc.) from the highlighted text and create a new ANSI code, keeping only the color attributes
 98		for code in codes:
 99			try:
100				text_attribute |= attr[code]
101			except KeyError:
102				ansi_code += (f'{code};')
103		ansi_code = ansi_code.rstrip(';')
104
105		if ansi_code:
106			colors.append((colorize_256.color_map[ansi_code] | text_attribute, len(substring_text)))
107		else:
108			colors.append((text_attribute, len(substring_text)))
109
110	return colors

164			self.parent.stored_highlights[self.values[i]] = [

167					self.parent.theme_manager.findPair(self, 'DEFAULT')
168				) for c in [color] * length
169			]

172		"""
173		Removes the **bold**, _italics_ and `code` markups from the input text.

175		Args:
176			text: The text to parse.
177
178		Returns:
179			The text stripped of the markup symbols.
180		"""
181		return re.sub(r'((\*\*|__)|(\*|_)|(\`))(.*?)\1', r'\5', text)

358	"""

360
361	Adapted from the [`npyscreen` library](
362		https://github.com/npcole/npyscreen/blob/8ce31204e1de1fbd2939ffe2d8c3b3120e93a4d0/npyscreen/utilNotify.py#L83

374	"""
375	message = _prepare_message(message)
376	if wide:
377		popup_form = YesNoPopupWide(name=title, color=form_color)
378	else:
379		popup_form = YesNoPopup(name=title, color=form_color)
380	popup_form.preserve_selected_widget = True
381
382	mlw = popup_form.add(Pager)
383	if wrap:
384		message = _wrap_message_lines(message, mlw.width - 1)
385	mlw.values = message
386
387	popup_form.editw = 0 #pylint: disable=attribute-defined-outside-init
388	popup_form.edit()
389
390	return popup_form.value
