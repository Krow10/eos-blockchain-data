176		jwt = get_auth_token()

193					f'The endpoint "{Config.GRPC_ENDPOINT}" doesn\'t provide a gRPC reflection service.\n\n'
194					f'Please select another one or fill the stub configuration file manually.',
195					title='Error: no reflection service available'
196				)
197				self.on_ok = lambda *args, **kwargs: self.parentApp.setNextFormPrevious()

199		self.ml_services = self.add(
200			TitleSelectOne,
201			name='Select a service',
202			values=services,
203			scroll_exit=True
204		)

211		# Currently detects if the endpoint is using substream by checking the name of the service
212		# TODO: Is this robust and reliable enough (?)
213		self.parentApp.is_substream = 'substream' in self.parentApp.selected_service
214
215		self.parentApp.stub_config['base'], self.parentApp.stub_config['service'] = \

320class StubConfigInputsForm(ActionFormV2, MarkdownEnabledHelpForm):
321	"""
322	Edit the request parameters sent to the gRPC endpoint.
323
324	Input options will be created according to their expected types (bool -> `InputBoolean`, etc.).
325
326	Attributes:
327		w_inputs: An `InputListDisplay` widget to present the list of input options.
328	"""
329	def clear_input(self, show_popup: bool = True) -> None:
330		"""
331		Callback function for clearing input shortcuts.
332
333		Pressing 'c' will ask for confirmation before clearing, 'C' will not.
334
335		Args:
336			show_popup: If True, asks the user for confirmation before clearing the input.
337		"""
338		input_widget = self.get_widget('inputs')
339		cleared_option = input_widget.values[input_widget.cursor_line]
340
341		clear_confirm = True
342		if show_popup:
343			clear_confirm = notify_yes_no(
344				'Are you sure you want to clear the value of this input ?',
345				title=f'Clear "{cleared_option.get_name_user()}" ?'
346			)
347
348		if clear_confirm:
349			input_widget.values[input_widget.cursor_line].set_from_widget_value('')
350			input_widget.display()
351
352	def create(self):
353		# Input clearing shortcuts, 'c' shows a warning, 'C' force clear the selected field
354		self.add_handlers({
355			'c': self.clear_input,
356			'C': lambda _: self.clear_input(show_popup=False)
357		})
358
359		# Map the corresponding cpp types to their input type implementation
360		cpptype_simplify_mapping = {
361			FieldDescriptor.CPPTYPE_INT32: 'Integer',
362			FieldDescriptor.CPPTYPE_INT64: 'Integer',
363			FieldDescriptor.CPPTYPE_UINT32: 'Integer',
364			FieldDescriptor.CPPTYPE_UINT64: 'Integer',
365			FieldDescriptor.CPPTYPE_DOUBLE: 'Float',
366			FieldDescriptor.CPPTYPE_FLOAT: 'Float',
367			FieldDescriptor.CPPTYPE_BOOL: 'Bool',
368			FieldDescriptor.CPPTYPE_ENUM: 'Enum',
369			FieldDescriptor.CPPTYPE_STRING: 'String',
370			FieldDescriptor.CPPTYPE_MESSAGE: 'Message'
371		}
372		# Reference :
373		# https://googleapis.dev/python/protobuf/latest/google/protobuf/descriptor.html#google.protobuf.descriptor.FieldDescriptor.CPPTYPE_BOOL
374
375		options = OptionList().options
376		for input_parameter in [
377			f for f in self.parentApp.selected_method.input_type.fields if not f.name in ('start_block_num', 'stop_block_num')
378		]:
379			# Load config value from loaded stub if it exists
380			try:
381				stub_config_value = self.parentApp.stub_config['request']['params'][input_parameter.name]
382			except KeyError:
383				stub_config_value = None
384
385			# Get the input type from the mapping (e.g. 'CPPTYPE_BOOL' -> 'Bool')
386			input_type = cpptype_simplify_mapping[input_parameter.cpp_type]
387
388			# Set the option type to the appropriate `InputXXX` class (e.g. 'InputBool')
389			option_type = getattr(input_options, f'Input{input_type}')
390			option_args = {
391				'documentation': [
392					f'A parameter of type {input_type.upper()} '
393					f'is expected for "{input_parameter.name}".'
394				],
395				'name': input_parameter.name,
396				'value': stub_config_value
397			}
398
399			# If its a repeated field, change to `InputRepeated` and pass the original type to the constructor
400			if input_parameter.label == FieldDescriptor.LABEL_REPEATED:
401				# TODO: Make repeated enum fields use multi-choice checkboxes
402				option_type = getattr(input_options, 'InputRepeated')
403				option_args.update(
404					# Allow the `InputRepeated` to pick the right validator (e.g. `bool_validator`)
405					value_type=input_type.lower(),
406				)
407
408			# TODO: Handle `oneof` option for FirehoseV2 `Fetch`
409
410			# Add or modify arguments based on the input type (`documentation`, etc.)
411			if input_parameter.cpp_type == FieldDescriptor.CPPTYPE_BOOL:
412				option_args.update(
413					documentation=option_args['documentation'] + [
414						'Press [X] or [SPACE] to toggle between checked/unchecked.'
415					]
416				)
417			elif input_parameter.cpp_type == FieldDescriptor.CPPTYPE_ENUM:
418				enum_choices = [e.name for e in input_parameter.enum_type.values]
419				option_args.update(
420					documentation=option_args['documentation'] + [
421						f'Valid values are {enum_choices}.'
422					],
423					choices=enum_choices
424				)
425			elif input_parameter.cpp_type == FieldDescriptor.CPPTYPE_MESSAGE:
426				if input_parameter.name.lower() == 'modules': # TODO: Add Firehose V2 `Fetch` message parsing
427					option_type = getattr(input_options, 'InputPackage')
428					option_args.update(
429						documentation=option_args['documentation'] + [
430							'Select a package file (.spkg) associated with the substream you want to use.'
431						],
432						parent=self
433					)
434			elif input_parameter.cpp_type == FieldDescriptor.CPPTYPE_STRING:
435				if self.parentApp.is_substream:
436					# Skip deprecated `output_modules` input parameter, not allowing multiple output modules
437					# See: https://github.com/streamingfast/substreams/blob/develop/docs/release-notes/change-log.md#single-module-output
438					if input_parameter.name.lower() == 'output_modules':
439						continue
440
441					if input_parameter.name.lower() == 'output_module':
442						package_modules = self.get_output_module_choices(
443							self.parentApp.stub_config.get('request', {}).get('params', {}).get('modules')
444						)
445						option_type = input_options.InputEnum
446						option_args.update(
447							documentation=[
448								'A parameter of type ENUM is expected for "output_module".',
449								f'Valid values are {package_modules}.'
450							],
451							choices=package_modules,
452							value=[stub_config_value]
453						)
454
455			# Add the new input instance to the list of options
456			options.append(option_type(**option_args))
457
458		self.w_inputs = self.add(
459			getattr(input_options, 'InputListDisplay'),
460			w_id='inputs',
461			name='Edit method inputs',
462			values=options,
463			scroll_exit=True
464		)
465
466		if self.parentApp.is_substream and not package_modules:
467			self.hide_input_option('output_module')
