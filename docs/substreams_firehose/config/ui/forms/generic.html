 20class MarkdownEnabledHelpForm(_FormBase):
 21	"""
 22	Allow markdown support for the help text and fix a display bug when returning from the help message display.
 23
 24	Attributes:
 25		help: The help text to display (can contain markdown markups to render).
 26	"""
 27	def __init__(self, *args, markdown_support: bool = True, **kwargs):
 28		super().__init__(*args, **kwargs)
 29
 30		if markdown_support and self.help:
 31			def h_display_help(_input: int) -> bool:
 32				flushinp()
 33				view_help(self.help, f'{self.name} | Help', autowrap=self.WRAP_HELP)
 34				return True
 35
 36			# Overwrite previous help handlers with the new one
 37			self.handlers = {
 38				KEY_F1: h_display_help,
 39				"KEY_F(1)": h_display_help,
 40				"^O": h_display_help,
 41			}
 42
 43	def display(self, clear: bool = True) -> None:
 44		"""
 45		Overwrite `display` method to prevent widgets showing blank after returning from the help message display.
 46
 47		See [`h_display_help()`](
 48			https://github.com/npcole/npyscreen/blob/8ce31204e1de1fbd2939ffe2d8c3b3120e93a4d0/build/lib/npyscreen/fmForm.py#L208
 49		) in `npyscreen` code for reference.
 50		"""
 51		super().display(clear)

114class CategorizedItemDisplayForm(ActionFormDiscard, MarkdownEnabledHelpForm): #pylint: disable=too-many-instance-attributes
115	"""
116	Display a list of items identified by a unique identifer, grouped by categories.
117
118	Selecting an item will bring up an `ActionButtonPopup` menu to edit or delete the item.
119	New items can be added using the third `ActionForm` button at the bottom (`New`).
120
121	Attributes:
122		items: A container of items to display. Items must have *gettable* attributes.
123		item_fields: A list of `ItemField` describing the fields of an item for edition.
124		identifier_key: An attribute of an item uniquely identifying this item from the others. This value will also be used for display.
125		category_key: An attribute items share for grouping them by category (set of unique values of that attribute).
126		default_category: The default category for items that don't have the `category_key` attribute.
127		sort_category: An optional function used to sort the categories (alphabetical by default).
128	"""
129	@dataclass
130	class ItemField:
131		"""
132		Local class to store an item field values.
133
134		Attributes:
135			name: The field name.
136			input_class: The input class from `inputs.py` to use for editing the field.
137			input_args: A dictionary with additional arguments to the `input_class`.
138			required: A boolean setting the field as required or optional.
139			documentation: A list of strings describing the field purpose.
140		"""
141		name: str
142		input_class: Type[InputGeneric]
143		input_args: dict = field(default_factory=dict)
144		required: bool = False
145		documentation: list[str] = field(default_factory=list)

147	class _CategorizedItemEditForm(ActionFormV2, MarkdownEnabledHelpForm):
148		"""
149		Edit an item fields.
150
151		Fields are marked as required (won't allow empty values) or optional, with different display (color and symbol) for each.
152		"""
153		REQUIRED_FIELD_COLOR = 'WARNING'
154		OPTIONAL_FIELD_COLOR = 'GOOD'
155
156		# Using string type hinting to allow forward declarations
157		def __init__(self, *args,
158			item: MutableMapping,
159			item_fields: Sequence['CategorizedItemDisplayForm.ItemField'],
160			parent: 'CategorizedItemDisplayForm',
161			**kwargs
162		):
163			self._item = item
164			self._item_fields = item_fields
165			self._items_form = parent
166			super().__init__(*args, **kwargs)
167
168		def create(self):
169			options = OptionList().options
170			for item_field in self._item_fields:
171				input_color = self.REQUIRED_FIELD_COLOR if item_field.required else self.OPTIONAL_FIELD_COLOR
172
173				options.append(item_field.input_class(
174					name=item_field.name,
175					value=self._item.get(item_field.name, ''),
176					documentation=\
177						[f'{"="*16}', f'=== {"REQUIRED" if item_field.required else "OPTIONAL"} ===', f'{"="*16}'] + item_field.documentation,
178					# Set the color inside the option editing window to match the displayed color from the fields list
179					option_widget_keywords={'labelColor': input_color},
180					**item_field.input_args
181				))
182
183				# Additional `Option` object parameters for display
184				options[-1].annotation_color = input_color
185				options[-1].required = item_field.required
186
187			self.w_inputs = self.add(
188				InputListDisplay,
189				w_id='inputs',
190				# TODO: Add name describing item types as help text ?
191				name='Edit item attributes',
192				values=options,
193				scroll_exit=True
194			)
195
196		def on_ok(self):
197			previous_id = self._item.get(self._items_form.identifier_key)
198
199			for item_field in self.w_inputs.values:
200				if item_field.required and not item_field.value:
201					notify_confirm(f'A value is required for "{item_field.name}"', title='Error: no value set for required field')
202					return
203
204				# Flag duplicate ids only when renaming
205				if item_field.name == self._items_form.identifier_key \
206				and item_field.value != previous_id \
207				and not self._items_form.is_unique(item_field.value):
208					notify_confirm(
209						f'The "{item_field.value}" id already exists. Please choose another identifier.',
210						title='Error: identifier not unique'
211					)
212					return
213
214				# Get the value from single choice widgets (stored internally as a list in the widget)
215				try:
216					item_field.value = item_field.value.pop()
217				except (AttributeError, IndexError):
218					pass
219
220				# Check if not empty or a boolean
221				is_empty_input = not (item_field.value or isinstance(item_field.value, bool))
222
223				# Delete any previously set parameter if it's empty or ignore it.
224				if item_field.name in self._item and is_empty_input:
225					del self._item[item_field.name]
226				elif not is_empty_input:
227					self._item[item_field.name] = item_field.value
228
229			# Update the items list display to move the edited item to its new corresponding `BoxTitle`
230			self._items_form.move_to_boxtitle(self._item)
231			self._items_form.select_item(self._item)
232
233			self.parentApp.setNextFormPrevious()
234
235		def on_cancel(self):
236			self.parentApp.setNextFormPrevious()

259	def create(self):
260		self.w_items_boxtitle = []
261
262		# Create a `BoxTitle` widget for each unique values of the `category` attribute
263		categories = self.sort_categories({entry.get(self.category_key, self.default_category) for entry in self.items})
264		n_items = len(categories)
265		for category in categories:
266			self.w_items_boxtitle.append(self.add(
267				CategorizedItemViewerBoxTitle,
268				name=category,
269				values=sorted(
270					[entry for entry in self.items if entry.get(self.category_key, self.default_category) == category],
271					key=lambda v: v.get(self.identifier_key)
272				),
273				max_height=self.lines//n_items - 4,
274				scroll_exit=True,
275			))

301	def on_ok(self) -> list[MutableMapping]:
302		"""
303		Hook for the `OK` button to return all the updated item values.
304
305		This can be used in overloaded methods to perform additional checks, transforms or store the updated data.
306
307		Returns:
308			The list of updated items.
309		"""
310		new_items = []

312			new_items.extend(boxtitle.values)

314		self.parentApp.setNextFormPrevious()
315		return new_items

321		"""
322		Override the default 'discard' behavior to create an new item entry.

324		self.create_item_edit_form(type(next(iter(self.items), None))())

327	def is_unique(self, item_id: Any) -> bool:
328		"""
329		Check that the item identified by the `item_id` is not already present in the displayed values.
330
331		Args:
332			item_id: An identifier unique to each item.
333
334		Returns:
335			A boolean indicating if the identifier is already present.
336
337		Raises:
338			ValueError: If the `item_id` is `None`.
339		"""
340		if item_id is None:
341			raise ValueError('Item id cannot be `None`')
342
343		return item_id not in [v.get(self.identifier_key) for w in self.w_items_boxtitle for v in w.values]
344
345	def move_to_boxtitle(self, item: MutableMapping) -> None:
346		"""
347		Remove the specified item from its previous `BoxTitle` and add it to the new one defined in its category attribute.
348
349		Args:
350			item: The item to move.
351		"""
352		for boxtitle_widget in self.w_items_boxtitle:
353			# Try to remove the item first
354			boxtitle_widget.values = [v for v in boxtitle_widget.values if v.get(self.identifier_key) != item.get(self.identifier_key)]
355
356			# Add it to the appropriate category
357			if item.get(self.category_key, self.default_category) == boxtitle_widget.name:
358				boxtitle_widget.values.append(item)
359				# Sort the entries back
360				boxtitle_widget.values = sorted(boxtitle_widget.values, key=lambda v: v.get(self.identifier_key))
361
362
363	def select_item(self, item: MutableMapping) -> None:
364		"""
365		Move the cursor to the specified item in the form. Used when editing is finished to select the edited item.
366
367		Args:
368			item: The item to select.
369		"""
370		for boxtitle_widget in self.w_items_boxtitle:
371			if item.get(self.category_key, self.default_category) == boxtitle_widget.name:
372				try:
373					# Update cursor position within the `BoxTitle` widget
374					boxtitle_widget.entry_widget.cursor_line = boxtitle_widget.values.index(item)
375
376					# Update the focused `BoxTitle` widget within the form
377					self.editw = self.w_items_boxtitle.index(boxtitle_widget) #pylint: disable=attribute-defined-outside-init
378				except ValueError:
379					logging.error('[%s] Item not found in %s widget : %s', self.name, boxtitle_widget.name, item)
380					raise
